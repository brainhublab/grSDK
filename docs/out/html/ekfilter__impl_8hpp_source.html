205  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
206  void EKFilter<T, BEG, OQ, OVR, DBG>::timeUpdateStep(Vector& u_) {
207 
208  // verif : u_.size() == nu
209  K_UINT_32 i, j, k;
210 
211  sizeUpdate();
212  u.swap(u_);
213 
214  makeCommonProcess();
215  makeAImpl();
216  makeWImpl();
217  makeQImpl();
218  makeProcess();
219 
220  if (!OQ) {
221 
222  if (flags & KALMAN_Q_MODIFIED) {
223 
224  Q_ = Q;
225  factor(Q_);
226  upperInvert(Q_);
227 
228  }
229 
230  Q.swap(Q_);
231 
232  // W_ = W*U n.nw = n.nw * nw.nw
233 
234  if (flags & ( KALMAN_W_MODIFIED | KALMAN_Q_MODIFIED ) ) {
235 
236  for (i = BEG; i < n + BEG; ++i) {
237 
238  for (j = BEG; j < nw + BEG; ++j) {
239 
240  W_(i,j) = W(i,j);
241  for (k = BEG; k < j; ++k)
242  W_(i,j) += W(i,k)*Q(j,k);
243 
244  }
245 
246  }
247 
248  }
249 
250  W.swap(W_);
251 
252  }
253 
254  timeUpdate();
255 
256  if (!OQ) {
257  Q.swap(Q_);
258  W.swap(W_);
259  }
260 
261  u.swap(u_);
262  flags &= ~KALMAN_MIDMASK;
263  }

265  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
266  void EKFilter<T, BEG, OQ, OVR, DBG>::measureUpdateStep(const Vector& z_) {
267 
268  // verif : z_.size() == m
269  K_UINT_32 i, j, k;
270 
271  sizeUpdate();
272 
273  if (m == 0) {
274  return;
275  }
276 
277  makeCommonMeasure();
278  makeHImpl();
279  makeVImpl();
280  makeRImpl();
281  makeMeasure();
282 
283  // verif : nv != 0
284 
285  for (i = BEG; i < m + BEG; ++i)
286  dz(i) = z_(i) - z(i);
287 
288  makeDZ();
289 
290  if (OVR) {
291 
292  // verif : m == nv
293 
294  if (flags & ( KALMAN_V_MODIFIED | KALMAN_R_MODIFIED ) ) {
295 
296  for (i = BEG; i < m + BEG; ++i)
297  R_(i,i) = V(i,i)*V(i,i)*R(i,i);
298 
299  }
300 
301  } else {
302 
303 
304  if (flags & ( KALMAN_V_MODIFIED | KALMAN_R_MODIFIED ) ) { // calculate R_
305 
306  _x.resize(nv);
307 
308  // R_ = V*R*V'
309  for (i = BEG; i < m + BEG; ++i) {
310 
311  // _x = row i of V*R = (V*R)(i,:)
312  for (j = BEG; j < nv + BEG; ++j) {
313 
314  _x(j) = T(0.0);
315  for (k = BEG; k < nv + BEG; ++k)
316  _x(j) += V(i,k)*R(k,j);
317 
318  }
319 
320  // R_(i,:) = (V*R*V')(i,:) = (V*R)(i,:) * V'
321  for (j = BEG; j < m + BEG; ++j) {
322 
323  R_(i,j) = T(0.0);
324  for (k = BEG; k < nv + BEG; ++k)
325  R_(i,j) += _x(k)*V(j,k);
326 
327  }
328 
329  }
330 
331  // R_ = U*D*U'
332  // diag(R_) = D, upper(R_) = U, lower(R_) = junk
333  factor(R_);
334 
335  // lower(R_) = (inv(U))'
336  upperInvert(R_);
337 
338  }
339 
340  if (flags & ( KALMAN_H_MODIFIED | KALMAN_V_MODIFIED | KALMAN_R_MODIFIED ) ) { // calculate H_
341 
342  // H_ = inv(U)*H m.n = m.m * m.n
343  for (i = BEG; i < m + BEG; ++i) {
344 
345  for (j = BEG; j < n + BEG; ++j) {
346 
347  H_(i,j) = H(i,j);
348  for (k = i + 1; k < m + BEG; ++k)
349  H_(i,j) += R_(k,i)*H(k,j);
350 
351  }
352 
353  }
354 
355  }
356 
357  H.swap(H_);
358 
359  // _x = inv(U)*dz m.1 = m.m * m.1
360  _x.resize(m);
361 
362  for (i = BEG; i < m + BEG; ++i) {
363 
364  _x(i) = dz(i);
365  for (k = i + 1; k < m + BEG; ++k)
366  _x(i) += R_(k,i)*dz(k);
367 
368  }
369 
370  dz.swap(_x);
371 
372  }
373 
374  _x.resize(n); // dx : innovation
375  _x = T(0.0);
376  for (i = BEG; i < m + BEG; ++i) {
377 
378  for (j = BEG; j < n + BEG; ++j)
379  a(j) = H(i,j);
380 
381  measureUpdate(dz(i), R_(i,i));
382 
383  }
384  for (i = BEG; i < n + BEG; ++i)
385  x(i) += _x(i);
386 
387  if (!OVR) {
388  H.swap(H_);
389  }
390 
391  flags &= ~KALMAN_HIGHMASK;
392  }

534  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
535  void EKFilter<T, BEG, OQ, OVR, DBG>::sizeUpdate() {
536 
537  if (!flags) {
538  return;
539  }
540 
541  if (flags & KALMAN_N_MODIFIED) {
542  A.resize(n, n);
543  makeBaseAImpl();
544  }
545 
546  if (flags & (KALMAN_N_MODIFIED | KALMAN_NW_MODIFIED) ) {
547  nn = n + nw;
548  a.resize(nn);
549  v.resize(nn);
550  d.resize(nn);
551  if (!OQ)
552  W_.resize(n, nw);
553  W.resize(n, nw);
554  makeBaseWImpl();
555  }
556 
557  // KALMAN_N_MODIFIED imply KALMAN_P_MODIFIED
558  // => KALMAN_N_MODIFIED must not be set OR KALMAN_P_MODIFIED must be set
559  // => NOT KALMAN_N_MODIFIED OR KALMAN_P_MODIFIED must be set
560  // verif : (flags ^ KALMAN_N_MODIFIED) &
561  // (KALMAN_N_MODIFIED | KALMAN_P_MODIFIED)
562 
563  if (flags & KALMAN_P_MODIFIED) {
564  // this covers the case of KALMAN_N_MODIFIED = true also
565 
566  // We have a new matrix P : let's factorize it and store it in U
567  // First, resize U and copy P in its left part
568  U.resize(n, nn);
569  for (K_UINT_32 i = BEG; i < n + BEG; ++i)
570  for (K_UINT_32 j = BEG; j < n + BEG; ++j)
571  U(i,j) = _P(i,j);
572 
573  // Factorize
574  factor(U);
575 
576  } else if (flags & KALMAN_NW_MODIFIED) {
577  // KALMAN_N_MODIFIED is necessarily false, else KALMAN_P_MODIFIED
578  // would have been true
579 
580  // Let's just copy U in temporary matrix _P of the right size,
581  // then swap the matrices
582  _P.resize(n, nn);
583  for (K_UINT_32 i = BEG; i < n + BEG; ++i)
584  for (K_UINT_32 j = i; j < n + BEG; ++j)
585  _P(i,j) = U(i,j);
586  U.swap(_P);
587  }
588 
589  if (flags & KALMAN_NW_MODIFIED) {
590  if (!OQ)
591  Q_.resize(nw, nw);
592  Q.resize(nw, nw);
593  makeBaseQImpl();
594  }
595 
596  if (m != 0) {
597 
598  if (flags & (KALMAN_N_MODIFIED | KALMAN_M_MODIFIED) ) {
599  if (!OVR)
600  H_.resize(m, n);
601  H.resize(m, n);
602  makeBaseHImpl();
603  }
604 
605  if (flags & (KALMAN_M_MODIFIED | KALMAN_NV_MODIFIED) ) {
606  V.resize(m, nv);
607  makeBaseVImpl();
608  }
609 
610  if (flags & KALMAN_NV_MODIFIED) {
611  R.resize(nv, nv);
612  makeBaseRImpl();
613  }
614 
615  if (flags & KALMAN_M_MODIFIED) {
616  R_.resize(m, m);
617  z.resize(m);
618  dz.resize(m);
619  }
620 
621  }
622 
623  flags &= ~KALMAN_LOWMASK;
624  }

626  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
627  void EKFilter<T, BEG, OQ, OVR, DBG>::factor(Matrix& P_) {
628 
629  // ne pas vérifier que P_.ncol() == P_.nrow(), comme ça, même si
630  // nrow() < ncol(), on peut factoriser la sous-matrice carrée de P
631  // Utile pour factoriser U
632 
633  T alpha, beta;
634  K_UINT_32 i, j, k, N = P_.nrow();
635  for(j = N - 1 + BEG; j > BEG; --j) {
636  alpha = T(1.0)/P_(j,j);
637  for(k = BEG; k < j; ++k) {
638  beta = P_(k,j);
639  P_(k,j) = alpha*beta;
640  for(i = BEG; i <= k; ++i)
641  P_(i,k) -= beta*P_(i,j);
642  }
643  }
644  }

646  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
647  void EKFilter<T, BEG, OQ, OVR, DBG>::upperInvert(Matrix& P_) {
648 
649  T val;
650  K_UINT_32 i, j, k, N = P_.nrow();
651  for (i = N - 2 + BEG; i != (K_UINT_32)(BEG-1); --i) { // intended overflow if BEG==0
652  for (k = i + 1; k < N + BEG; ++k) {
653 
654  val = P_(i,k);
655  for (j = i + 1; j <= k - 1; ++j)
656  val += P_(i,j)*P_(k,j);
657  P_(k,i) = -val;
658 
659  }
660  }
661 
662  }

664  // U u U-D covariance matrix (n,nn)
665  // A phi transition matrix (F) (n,n)
666  // W g process noise matrix (G) (n,nw)
667  // Q q process noise variance vector (nw) Q = diag(q)
668  // a, v, d temporary vectors
669  // U is updated
670  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
671  void EKFilter<T, BEG, OQ, OVR, DBG>::timeUpdate() {
672 
673  K_UINT_32 i, j, k;
674  T sigma, dinv;
675 
676  // U = phi * U
677  // d = diag(U)
678  //
679  // This algo could be faster
680  // if phi is known to be diagonal
681  // It could be almost zapped if phi=I
682  for(j = n - 1 + BEG; j > BEG; --j) {
683  for(i = BEG; i <= j; ++i)
684  d(i) = U(i,j);
685  for(i = BEG; i < n + BEG; ++i) {
686  U(i,j) = A(i,j);
687  for(k = BEG; k < j; ++k)
688  U(i,j) += A(i,k)*d(k);
689  }
690  }
691 
692  d(BEG) = U(BEG,BEG);
693  for(j = BEG; j < n + BEG; ++j)
694  U(j,BEG) = A(j,BEG);
695 
696  // d(n+1:nn) = q
697  // U(:,n+1:nn) = G
698  for(i = BEG; i < nw + BEG; ++i) {
699  d(i+n) = Q(i,i);
700  for(j = BEG; j < n + BEG; ++j)
701  U(j,i+n) = W(j,i);
702  }
703 
704  // Gram-Schmidt
705  // Too hard to simplify
706  for(j = n - 1 + BEG; j != (K_UINT_32)(BEG-1); --j) { // intended overflow if BEG==0
707  sigma = T(0.0);
708  for(k = BEG; k < nn + BEG; ++k) {
709  v(k) = U(j,k);
710  a(k) = d(k)*v(k);
711  sigma += v(k)*a(k);
712  }
713  U(j,j) = sigma;
714  if(j == BEG || sigma == T(0.0)) continue;
715  dinv = T(1.0)/sigma;
716  for(k = BEG; k < j; ++k) {
717  sigma = T(0.0);
718  for(i = BEG; i < nn + BEG; ++i)
719  sigma += U(k,i)*a(i);
720  sigma *= dinv;
721  for(i = BEG; i < nn + BEG; ++i)
722  U(k,i) -= sigma*v(i);
723  U(j,k) = sigma;
724  }
725  }
726 
727  // U = transpose(U)
728  for(j = BEG + 1; j < n + BEG; ++j)
729  for(i = BEG; i < j; ++i)
730  U(i,j) = U(j,i);
731  }
732 
733  // x a priori estimate vector (n)
734  // U a priori U-D covariance matrix (n,nn)
735  // dz measurement diff (z - ax) (scalar)
736  // a measurement coefficients vector (n) (a row of A, which is H)
737  // a is destroyed
738  // r measurement variance
739  // d is a temporary vector
740  // x and U are updated
741  // a is destroyed
742  template<typename T, K_UINT_32 BEG, bool OQ, bool OVR, bool DBG>
743  void EKFilter<T, BEG, OQ, OVR, DBG>::measureUpdate(T dz, T r) {
744 
745  K_UINT_32 i, j, k;
746  T alpha, gamma, beta, lambda;
747 
748  // dz = dz - Hdx
749  for (j = BEG; j < n + BEG; ++j)
750  dz -= a(j)*_x(j);
751 
752  // d = D * transpose(U) * a
753  // a = transpose(U) * a
754  //
755  // This algo could be faster
756  // if A is known to be diagonal or I
757  for(j = n - 1 + BEG; j > BEG; --j) {
758  for(k = BEG; k < j; ++k)
759  a(j) += U(k,j)*a(k);
760  d(j) = U(j,j)*a(j);
761  }
762  d(BEG) = U(BEG,BEG)*a(BEG);
763 
764  // UDU
765  // Too hard to simplify
766  alpha = r+d(BEG)*a(BEG);
767  gamma = T(1.0)/alpha;
768  U(BEG,BEG) = r*gamma*U(BEG,BEG);
769  for(j = BEG + 1; j < n + BEG; ++j) {
770  beta = alpha;
771  alpha += d(j)*a(j);
772  lambda = -a(j)*gamma;
773  gamma = T(1.0)/alpha;
774  U(j,j) *= beta*gamma;
775  for(i = BEG; i < j; ++i) {
776  beta = U(i,j);
777  U(i,j) = beta+d(i)*lambda;
778  d(i) += d(j)*beta;
779  }
780  }
781 
782  // dx = dx + K(dz - Hdx)
783  dz *= gamma;
784  for(j = BEG; j < n + BEG; ++j)
785  _x(j) += d(j)*dz;
786  }
