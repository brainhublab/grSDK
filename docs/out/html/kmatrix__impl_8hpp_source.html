103  template<typename T, K_UINT_32 BEG, bool DBG>
104  inline void KMatrix<T, BEG, DBG>::init(K_UINT_32 m, K_UINT_32 n) {
105 
106  if (m != 0 && n != 0) {
107 
108  // non-empty matrix
109  vimpl_.resize(m);
110  T* ptr = &Mimpl_[0] - BEG;
111  T** M = &vimpl_[0];
112  T** end = M + m;
113 
114  while (M != end) {
115  *M++ = ptr;
116  ptr += n;
117  }
118 
119  M_ = &vimpl_[0] - BEG;
120  m_ = m;
121  n_ = n;
122 
123  } else {
124 
125  // empty matrix
126  M_ = 0;
127  m_ = 0;
128  n_ = 0;
129 
130  }
131 
132  }
