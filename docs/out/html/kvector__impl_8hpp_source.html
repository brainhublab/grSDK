237  template<typename T, K_UINT_32 BEG, bool DBG>
238  inline void KVector<T, BEG, DBG>::get(std::istream& is) {
239  if (n_ == 0) {
240  return;
241  }
242 
243  T* ptr = &vimpl_[0];
244  std::string tmp;
245  K_UINT_32 i;
246 
247  if (currentVectorContext->skipStartDelim_) {
248  is >> tmp;
249  }
250 
251  if (currentVectorContext->skipElemDelim_) {
252  for (i = 0; i < n_-1; ++i) {
253  is >> ptr[i] >> tmp;
254  }
255  is >> ptr[i];
256  } else {
257  for (i = 0; i < n_; ++i) {
258  is >> ptr[i];
259  }
260  }
261 
262  if (currentVectorContext->skipEndDelim_) {
263  is >> tmp;
264  }
265  }
